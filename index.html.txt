function handleImg(input) {
  const file = input.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    document.getElementById('imgData').value = e.target.result;
    document.getElementById('imgBox').innerHTML = `<img src="${e.target.result}">`;
  };
  reader.readAsDataURL(file);
}

function saveProduct() {
  const name = document.getElementById('fName').value.trim();
  const price = document.getElementById('fPrice').value;
  if(!name||!price) { showToast('❌ Name aur price zaroori hain!','error'); return; }
  const eid = document.getElementById('editingId').value;
  const obj = {
    id: eid ? parseInt(eid) : Date.now(),
    name, desc: document.getElementById('fDesc').value,
    price: parseInt(price),
    oldPrice: document.getElementById('fOldPrice').value || null,
    category: document.getElementById('fCategory').value,
    badge: document.getElementById('fBadge').value,
    img: document.getElementById('imgData').value || ''
  };
  if(eid) { const i=products.findIndex(p=>p.id==eid); products[i]=obj; showToast('✅ Product update ho gaya!'); }
  else { products.push(obj); showToast('✅ Product add ho gaya!'); }
  save(); updateStats(); resetForm();
}

function resetForm() {
  document.getElementById('editingId').value='';
  ['fName','fDesc','fPrice','fOldPrice','imgData'].forEach(id=>document.getElementById(id).value='');
  document.getElementById('fileInput').value='';
  document.getElementById('imgBox').innerHTML='<span>📷</span><span>Click kar ke image upload karo</span>';
  document.getElementById('formHeading').textContent='➕ Add Product';
}

function openEdit(id) {
  const p=products.find(x=>x.id===id);
  if(!p) return;
  document.getElementById('eId').value=id;
  document.getElementById('eName').value=p.name;
  document.getElementById('eDesc').value=p.desc;
  document.getElementById('ePrice').value=p.price;
  document.getElementById('eOldPrice').value=p.oldPrice||'';
  document.getElementById('eCategory').value=p.category;
  document.getElementById('eBadge').value=p.badge;
  document.getElementById('editModal').classList.add('open');
}

function closeModal() { document.getElementById('editModal').classList.remove('open'); }

function saveEdit() {
  const id=parseInt(document.getElementById('eId').value);
  const i=products.findIndex(p=>p.id===id);

}
